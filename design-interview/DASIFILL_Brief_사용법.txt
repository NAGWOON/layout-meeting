        function selectOption(element) {
            // 기존 선택 해제
            document.querySelectorAll('.option-item').forEach(el => {
                el.classList.remove('selected');
            });
            // 클릭된 요소 선택
            element.classList.add('selected');
            
            // 꼬리 질문 영역과 다음 버튼 부드럽게 나타나기
